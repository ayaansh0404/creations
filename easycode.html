Install easyCode

Put this code in your HTML head to use easyCode:

<script>
fetch("https://gist.githubusercontent.com/ayaansh0404/575a9ebfef498b2bc557307b311a02c2/raw/52529a53bf1d2aa2313ff799e4ebc6167029f422/easycode.css")
.then(res => res.text())
.then(css => {
const style = document.createElement("style");
style.textContent = css;
document.head.appendChild(style);
});
</script>
